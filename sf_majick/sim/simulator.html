function rand(a, b) { return a + Math.random() * (b - a); }
function randInt(a, b) { return Math.floor(rand(a, b + 1)); }
function clamp(v, lo, hi) { return Math.max(lo, Math.min(hi, v)); }
function lognorm(mu, sigma) {
  const u = Math.random() + Math.random() + Math.random() - 1.5;
  return Math.exp(mu + sigma * u * 1.732);
}
function fmtK(v) {
  if (v >= 1e6) return (v/1e6).toFixed(1) + 'M';
  if (v >= 1e3) return (v/1e3).toFixed(0) + 'K';
  return Math.round(v).toString();
}

function sync(el, valId, fmt) {
  const v = parseFloat(el.value);
  const sp = document.getElementById(valId);
  if (!sp) return;
  if      (fmt === 'pct')  sp.textContent = Math.round(v * 100) + '%';
  else if (fmt === 'pct2') sp.textContent = (v * 100).toFixed(1) + '%';
  else if (fmt === 'pct3') sp.textContent = (v * 100).toFixed(2) + '%';
  else if (fmt === 'dec1') sp.textContent = v.toFixed(1);
  else if (fmt === 'dec2') sp.textContent = v.toFixed(2);
  else if (fmt === 'dec3') sp.textContent = v.toFixed(3);
  else if (fmt === 'x')    sp.textContent = v.toFixed(2) + '×';
  else                     sp.textContent = Math.round(v);
  // live-update the deal size preview if a revenue slider changed
  if (valId === 'v-rev-mu' || valId === 'v-rev-sig') updateRevPreview();
}

function pickWeighted(weights) {
  const tot = weights.reduce((a,b) => a+b, 0) || 1;
  let r = Math.random() * tot, cum = 0;
  for (let i = 0; i < weights.length; i++) {
    cum += weights[i];
    if (r <= cum) return i;
  }
  return weights.length - 1;
}

function simulateOrg(cfg) {
  const theta = {
    ...THETA,
    base_prob_close:             cfg.base_prob_close,
    base_prob_lost:              cfg.base_prob_lost,
    base_prob_prospecting:       cfg.base_prob_prospecting,
    base_prob_qualification:     cfg.base_prob_qualification,
    base_prob_proposal:          cfg.base_prob_proposal,
    base_prob_negotiation:       cfg.base_prob_negotiation,
    base_prob_lead_conversion:   cfg.base_prob_lead_conversion,
    inactivity_alpha:            cfg.inactivity_alpha,
    stagnation_alpha:            cfg.stagnation_alpha,
    sentiment_beta_opportunity:  cfg.sentiment_beta_opportunity,
    momentum_beta_opportunity:   cfg.momentum_beta_opportunity,
    friction_beta_opportunity:   cfg.friction_beta_opportunity,
  };

  // Build reps
  const archWeights = [cfg.closer, cfg.nurturer, cfg.grinder, cfg.scattered];
  const archNames   = ['Closer', 'Nurturer', 'Grinder', 'Scattered'];
  const reps = Array.from({length: cfg.n_reps}, (_, i) => {
    const arch = archNames[pickWeighted(archWeights)];
    const props = ARCHETYPE_PROPS[arch];
    return { id:`rep_${i}`, arch, earnings:0, stress:0, motivation:1.0,
             wins:0, losses:0, days_burned_out:0, lifetime_actions:0,
             comp_rate: cfg.comp_rate, ...props };
  });

  // Build accounts + opps
  const accPerWeights = [cfg.p_skeptic, cfg.p_urgent, cfg.p_price, cfg.p_easy];
  function mkOpp(i, repId) {
    // Revenue drawn from lognormal with user-controlled mu and sigma.
    // mu = ln(median deal size), so rev-mu=10.82 → median ~$50K.
    // No extra multiplier — the log-space parameters already determine the scale.
    const rev  = lognorm(cfg.rev_mu, cfg.rev_sigma);
    const diff = clamp(Math.random() * cfg.difficulty_mean * 2, 0.005, cfg.difficulty_mean * 3);
    const pIdx = pickWeighted(accPerWeights);
    const pers = ACCOUNT_PERSONALITIES[pIdx];
    return {
      id:`opp_${i}`, name:`Account_${i}`, rep_id:repId, stage:'Prospecting',
      revenue:rev, difficulty:diff,
      sentiment: (Math.random()-0.5)*0.4,
      sentiment_history:[], days_in_stage:0, days_since_touch:0,
      touched_today:false, is_closed:false,
      micro: Object.fromEntries(Object.keys(SENTIMENT_EFFECTS).map(k=>[k,0])),
      personality:pers.name, urgency:pers.urgency, skepticism:pers.skepticism,
      price_sensitivity:pers.price_sensitivity,
      close_day:null, outcome:null
    };
  }
  const opps = Array.from({length: cfg.n_accounts}, (_, i) => mkOpp(i, reps[i % reps.length].id));

  function meetsReqs(opp, stage) {
    const reqs = STAGE_ADV_REQS[stage] || {};
    return Object.entries(reqs).every(([a,n]) => (opp.micro[a]||0) >= n);
  }

  const dayStats = [];
  const allActions = Object.keys(SENTIMENT_EFFECTS);

  for (let day = 0; day < cfg.days; day++) {
    let dayActs = 0, dayWon = 0, dayLost = 0, dayAdv = 0;

    for (const rep of reps) {
      // Decay stress
      rep.stress = Math.max(0, rep.stress - rep.resilience * 4);
      const isBurning = rep.stress >= rep.burnout;
      if (isBurning) rep.days_burned_out++;

      let attn = isBurning ? 160 : 360;
      attn = Math.floor(attn * (0.60 + 0.40 * rep.motivation));
      if (Math.random() < rep.distraction) attn -= randInt(30, 80);

      const myOpps = opps.filter(o => o.rep_id === rep.id && !o.is_closed);
      if (!myOpps.length) continue;

      const nWork = Math.min(myOpps.length, Math.max(1, Math.floor(attn/110)));
      // Prioritize later stages
      myOpps.sort((a,b) => STAGE_ORDER.indexOf(b.stage) - STAGE_ORDER.indexOf(a.stage));
      const worked = myOpps.slice(0, nWork);

      for (const opp of worked) {
        opp.touched_today = true;
        opp.days_since_touch = 0;

        let oppAttn = Math.floor(attn / worked.length);
        const nActs = randInt(2, 5);
        for (let ai = 0; ai < nActs && oppAttn > 15; ai++) {
          // Personality-biased action selection
          const actionWeights = allActions.map(a => {
            let w = 1.0;
            if (['make_call','hold_meeting','send_proposal','stakeholder_alignment'].includes(a)) w *= 1 + rep.aggression;
            if (['send_email','follow_up','internal_prep'].includes(a)) w *= 1 + rep.empathy;
            if (['research_account','internal_prep','solution_design'].includes(a)) w *= 1 + rep.discipline;
            if (rep.discipline < 0.35 && ['send_email','make_call'].includes(a)) w *= 1.5;
            return w;
          });
          const action = allActions[pickWeighted(actionWeights)];
          const cost = ACTION_COSTS[action] || 20;
          if (oppAttn < cost) continue;

          // Aggression short-circuit
          if (['internal_prep','research_account','solution_design'].includes(action)
              && rep.aggression > 0.70 && Math.random() < (rep.aggression - 0.70)) continue;

          oppAttn -= cost;
          opp.micro[action]++;
          rep.lifetime_actions++;
          dayActs++;

          // Sentiment delta
          let delta = (SENTIMENT_EFFECTS[action] || 0.05) * (0.4 + Math.random() * 0.8);
          delta *= 1 + opp.urgency * 0.25 - opp.skepticism * 0.15 - opp.price_sensitivity * 0.05;
          if (rep.arch === 'Closer'   && ['hold_meeting','send_proposal'].includes(action)) delta *= 1.3;
          if (rep.arch === 'Nurturer' && ['follow_up','send_email','hold_meeting'].includes(action)) delta *= 1.2;
          if (rep.arch === 'Grinder') delta *= 1.05;
          opp.sentiment = clamp(opp.sentiment + delta, -5, 5);
          opp.sentiment_history.push(delta);
        }

        // Empathy bonus touch
        if (rep.empathy > 0.75 && Math.random() < (rep.empathy - 0.75) && oppAttn > 30) {
          opp.micro['follow_up'] = (opp.micro['follow_up']||0) + 1;
          opp.sentiment = clamp(opp.sentiment + 0.08, -5, 5);
        }

        // Stage advancement
        const idx = STAGE_ORDER.indexOf(opp.stage);
        if (idx >= 0 && idx < STAGE_ORDER.length - 1) {
          const nextStage = STAGE_ORDER[idx + 1];
          if (meetsReqs(opp, nextStage)) {
            const stageBase = [theta.base_prob_prospecting, theta.base_prob_qualification,
                               theta.base_prob_proposal, theta.base_prob_negotiation][idx];
            const momentum = opp.sentiment_history.slice(-5).reduce((a,b)=>a+b,0)/5;
            const friction = Math.max(0, -(opp.sentiment_history.slice(-5).reduce((a,b)=>a+b,0)/5));
            const diffScaled = opp.difficulty * Math.pow(opp.revenue / theta.revenue_base, theta.revenue_exponent);
            const p = clamp(stageBase
                            + momentum  * theta.momentum_beta_opportunity * 0.35
                            - friction  * theta.friction_beta_opportunity * 0.35
                            + opp.sentiment * theta.sentiment_beta_opportunity * 0.5
                            - diffScaled, 0.005, 0.50);
            if (Math.random() < p) {
              opp.stage = nextStage;
              opp.days_in_stage = 0;
              dayAdv++;
            }
          }
        }

        // Close check (Negotiation only)
        if (opp.stage === 'Negotiation' && meetsReqs(opp, 'Negotiation')) {
          const momentum = opp.sentiment_history.slice(-5).reduce((a,b)=>a+b,0)/5;
          const friction = Math.max(0, -(opp.sentiment_history.slice(-5).reduce((a,b)=>a+b,0)/5));
          const diffScaled = opp.difficulty * Math.pow(opp.revenue / theta.revenue_base, 0.05);
          const pClose = clamp(theta.base_prob_close
                               + momentum * theta.momentum_beta_opportunity * 0.40
                               - friction * theta.friction_beta_opportunity * 0.30
                               + opp.sentiment * theta.sentiment_beta_opportunity * 0.30
                               - diffScaled, 0.005, 0.70);
          if (Math.random() < pClose) {
            opp.stage = 'Closed Won'; opp.is_closed = true;
            opp.close_day = day; opp.outcome = 'won';
            const archBonus = rep.arch === 'Closer' ? 1.15 : 1.0;
            const commission = opp.revenue * cfg.commission_rate * rep.comp_rate * archBonus;
            rep.earnings += commission; rep.wins++;
            rep.motivation = Math.min(1.0, rep.motivation + 0.18);
            rep.stress = Math.max(0, rep.stress - 4);
            dayWon++;
            continue;
          }
        }

        // Lost check
        const pLost = clamp(
          theta.base_prob_lost
          + theta.inactivity_alpha * Math.log1p(opp.days_since_touch)
          + theta.stagnation_alpha * Math.log1p(opp.days_in_stage),
          0, 0.35
        );
        if (Math.random() < pLost) {
          opp.stage = 'Closed Lost'; opp.is_closed = true;
          opp.close_day = day; opp.outcome = 'lost';
          rep.losses++; rep.motivation = Math.max(0.1, rep.motivation - 0.05);
          rep.stress += 2.5;
          dayLost++;
        }
      }

      // Tick untouched
      for (const opp of myOpps) {
        if (!opp.touched_today && !opp.is_closed) {
          opp.days_since_touch++;
          opp.days_in_stage++;
        }
        opp.touched_today = false;
      }

      // Daily rep stress accumulation
      const workloadStress = 0.4 * dayActs / Math.max(worked.length, 1);
      rep.stress = Math.min(rep.stress + workloadStress, 200);
      rep.consecutive_no_close_days = (rep.consecutive_no_close_days || 0) + (dayWon ? 0 : 1);
      if (rep.consecutive_no_close_days > 14) {
        rep.motivation = Math.max(0.1, rep.motivation - 0.01);
      }
    }

    dayStats.push({day, actions:dayActs, advanced:dayAdv, won:dayWon, lost:dayLost});
  }

  // Compile
  const closed = opps.filter(o => o.is_closed);
  const won    = closed.filter(o => o.outcome === 'won');
  const lost   = closed.filter(o => o.outcome === 'lost');
  const wonRevenue  = won.reduce((s,o) => s + o.revenue, 0);
  const stageCount  = Object.fromEntries(STAGES.map(s => [s, opps.filter(o => o.stage === s).length]));
  const avgCycleDays = won.length ? won.reduce((s,o)=>s+(o.close_day||0),0)/won.length : null;
  const avgDealSize  = won.length ? wonRevenue / won.length : null;
  const winRate      = closed.length ? won.length / closed.length : null;
  const totalRevenue = wonRevenue;
  const totalActions = dayStats.reduce((s,d)=>s+d.actions,0);

  return { cfg, reps, opps, stageCount, won, lost, closed,
           wonRevenue, winRate, avgCycleDays, avgDealSize, dayStats,
           totalRevenue, totalActions };
}

function calibrate(d) {
  const opps = d.opps;
  const cfg  = d.cfg;

  // Win rate → base_prob_close
  const base_prob_close = d.winRate != null ? +clamp(d.winRate, 0.01, 0.60).toFixed(4) : null;

  // Lost rate → daily hazard (mirroring OrgCalibrator._compute_conversion_rates)
  const lostRate = opps.length > 0 ? d.lost.length / opps.length : null;
  const avgCycle = d.avgCycleDays || 60;
  const base_prob_lost = lostRate != null ? +clamp(lostRate / Math.max(avgCycle, 1), 0.001, 0.10).toFixed(5) : null;

  // Stage advancement rates (fraction of opps that advanced past each stage)
  function stageAdvRate(stage) {
    const stageIdx = STAGE_ORDER.indexOf(stage);
    const pastThis = opps.filter(o => {
      const oIdx = STAGE_ORDER.indexOf(o.stage);
      const finalIdx = o.is_closed && o.outcome === 'won' ? STAGE_ORDER.length : oIdx;
      return finalIdx > stageIdx;
    }).length;
    const total = opps.filter(o => STAGE_ORDER.indexOf(o.stage) >= stageIdx || (o.is_closed && o.outcome === 'won')).length;
    return total > 0 ? pastThis / total : null;
  }

  function rateToDaily(stage, rate) {
    if (rate == null) return null;
    const avgDays = {Prospecting:25, Qualification:20, Proposal:15, Negotiation:12}[stage] || 20;
    return +clamp(rate / Math.max(avgDays, 1), 0.001, 0.30).toFixed(4);
  }

  // Cycle time → inactivity/stagnation alphas
  const inactivity_alpha = +clamp(THETA.inactivity_alpha * (45 / Math.max(avgCycle, 10)), 0.002, 0.025).toFixed(5);
  const stagnation_alpha = +clamp(THETA.stagnation_alpha * (45 / Math.max(avgCycle, 10)), 0.001, 0.020).toFixed(5);

  // Archetype distribution
  const archCounts = {};
  for (const r of d.reps) archCounts[r.arch] = (archCounts[r.arch] || 0) + 1;
  const archetype_weights = {};
  for (const [k,v] of Object.entries(archCounts)) archetype_weights[k] = +(v / d.reps.length).toFixed(2);

  // Account personality distribution (observed from opps)
  const persCounts = {};
  for (const o of opps) persCounts[o.personality] = (persCounts[o.personality]||0)+1;
  const account_personality_weights = {};
  for (const [k,v] of Object.entries(persCounts)) account_personality_weights[k] = +(v/opps.length).toFixed(2);

  const prospAdv = stageAdvRate('Prospecting');
  const qualAdv  = stageAdvRate('Qualification');
  const propAdv  = stageAdvRate('Proposal');

  return {
    label: 'fitted_from_sim',
    n_reps: d.reps.length,
    archetype_weights,
    account_personality_weights,
    n_seed_accounts: cfg.n_accounts,
    n_seed_leads: cfg.n_leads,
    days: cfg.days,
    n_runs: 100,
    base_prob_close,
    base_prob_lost,
    base_prob_prospecting:   rateToDaily('Prospecting', prospAdv),
    base_prob_qualification: rateToDaily('Qualification', qualAdv),
    base_prob_proposal:      rateToDaily('Proposal', propAdv),
    inactivity_alpha,
    stagnation_alpha,
    apply_decay: true,
    notes: `Auto-fitted from ${cfg.days}-day simulation. ${d.won.length} Closed Won, ${d.lost.length} Closed Lost out of ${opps.length} total opps.`,
    _observed: {
      win_rate:       d.winRate != null ? +(d.winRate*100).toFixed(1) : null,
      avg_cycle_days: d.avgCycleDays != null ? +d.avgCycleDays.toFixed(1) : null,
      avg_deal_size:  d.avgDealSize != null ? +d.avgDealSize.toFixed(0) : null,
      total_actions:  d.totalActions,
      stage_adv_rates: {
        Prospecting:   prospAdv != null ? +(prospAdv*100).toFixed(1) : null,
        Qualification: qualAdv  != null ? +(qualAdv *100).toFixed(1) : null,
        Proposal:      propAdv  != null ? +(propAdv *100).toFixed(1) : null,
      }
    }
  };
}

// --- Simple CSV parser (handles quoted fields) ---
function parseCsv(text) {
  const lines = text.trim().split(/\r?\n/);
  if (lines.length < 2) return [];
  const headers = splitCsvLine(lines[0]).map(h => h.toLowerCase().trim().replace(/^"|"$/g,''));
  return lines.slice(1).filter(l => l.trim()).map(line => {
    const vals = splitCsvLine(line);
    const row = {};
    headers.forEach((h, i) => { row[h] = (vals[i] || '').replace(/^"|"$/g,'').trim(); });
    return row;
  });
}

function splitCsvLine(line) {
  const result = []; let cur = ''; let inQ = false;
  for (let i = 0; i < line.length; i++) {
    const c = line[i];
    if (c === '"') { inQ = !inQ; }
    else if (c === ',' && !inQ) { result.push(cur); cur = ''; }
    else cur += c;
  }
  result.push(cur);
  return result;
}

// --- Find best matching column ---
function detectCol(headers, key) {
  const aliases = COL_ALIASES[key] || [key];
  for (const a of aliases) {
    const match = headers.find(h => h === a || h.replace(/\s/g,'') === a.replace(/\s/g,''));
    if (match) return match;
  }
  return null;
}

// --- Core calibration logic (mirrors OrgCalibrator.calibrate()) ---
function calibrateFromCsv() {
  const opps = csvData.opps;
  const users = csvData.users;
  const leads = csvData.leads;
  const colMap = buildColMap();
  const warnings = [];

  function col(row, key) {
    const c = colMap[key];
    return c ? row[c] : undefined;
  }

  // Normalise stage
  const stageCol = colMap.stage;
  const oppsWithStage = opps.map(r => ({
    ...r,
    _stage_sim: STAGE_MAP[(col(r,'stage')||'').toLowerCase().trim()] || null,
    _amount: parseFloat(col(r,'amount')) || null,
    _close_date: col(r,'close_date') ? new Date(col(r,'close_date')) : null,
    _created_date: col(r,'created_date') ? new Date(col(r,'created_date')) : null,
    _owner: col(r,'owner_id') || null,
    _account: col(r,'account_id') || null,
  }));

  if (!stageCol) warnings.push({ level:'warn', msg: 'No stage column found — stage-based fitting will be skipped.' });

  // ---- Team size ----
  let n_reps = null;
  if (users && users.length > 0) {
    n_reps = users.length;
  } else {
    const owners = new Set(oppsWithStage.map(r => r._owner).filter(Boolean));
    if (owners.size > 0) n_reps = owners.size;
    else warnings.push({ level:'warn', msg: 'No owner/rep column — rep count not fitted, using default.' });
  }

  // ---- Accounts ----
  const accounts = new Set(oppsWithStage.map(r => r._account).filter(Boolean));
  const n_seed_accounts = accounts.size || oppsWithStage.length;

  // ---- Conversion rates ----
  const staged = oppsWithStage.filter(r => r._stage_sim);
  const nTotal = staged.length || oppsWithStage.length;
  const closed = staged.filter(r => r._stage_sim === 'Closed Won' || r._stage_sim === 'Closed Lost');
  const nClosed = closed.length;
  const nWon  = closed.filter(r => r._stage_sim === 'Closed Won').length;
  const nLost = closed.filter(r => r._stage_sim === 'Closed Lost').length;

  if (!staged.length) warnings.push({ level:'error', msg: 'Stage column found but no rows matched known SF stage names. Check the stage map or your column selection.' });

  const win_rate  = nClosed > 0 ? nWon  / nClosed : null;
  const lost_rate = nClosed > 0 ? nLost / nClosed : null;
  const base_prob_close = win_rate != null ? +clamp(win_rate, 0.01, 0.60).toFixed(4) : null;

  if (nClosed < 10) warnings.push({ level:'warn', msg: `Only ${nClosed} closed deals — win rate estimate may be unreliable. Need at least 10–20 for a stable fit.` });

  // ---- Cycle time ----
  const cycleRows = oppsWithStage.filter(r => r._close_date && r._created_date && r._close_date > r._created_date);
  let avg_cycle = null;
  if (cycleRows.length > 0) {
    const days = cycleRows.map(r => (r._close_date - r._created_date) / 86400000).filter(d => d > 0);
    days.sort((a,b) => a-b);
    avg_cycle = days[Math.floor(days.length/2)]; // median
  } else {
    warnings.push({ level:'warn', msg: 'No close_date / created_date columns — cycle time not fitted, inactivity/stagnation alphas use defaults.' });
  }

  const base_prob_lost = (lost_rate != null && avg_cycle)
    ? +clamp(lost_rate / Math.max(avg_cycle, 1), 0.001, 0.10).toFixed(5)
    : null;

  const inactivity_alpha = avg_cycle ? +clamp(0.010 * (45 / Math.max(avg_cycle, 10)), 0.002, 0.025).toFixed(5) : null;
  const stagnation_alpha = avg_cycle ? +clamp(0.008 * (45 / Math.max(avg_cycle, 10)), 0.001, 0.020).toFixed(5) : null;

  // ---- Stage advancement rates ----
  const STAGE_ORDER = ['Prospecting','Qualification','Proposal','Negotiation'];
  const stageCounts = {};
  for (const s of [...STAGE_ORDER, 'Closed Won', 'Closed Lost']) {
    stageCounts[s] = staged.filter(r => r._stage_sim === s).length;
  }
  // Include Closed Won as having "passed" all stages
  const wonCount = stageCounts['Closed Won'] || 0;

  function advRate(stage) {
    const idx = STAGE_ORDER.indexOf(stage);
    if (idx < 0) return null;
    // Opps "at or past" this stage = at this stage + all later stages + closed won/lost
    const atOrPast = staged.filter(r => {
      const ri = STAGE_ORDER.indexOf(r._stage_sim);
      return ri >= idx || r._stage_sim === 'Closed Won' || r._stage_sim === 'Closed Lost';
    }).length;
    const past = staged.filter(r => {
      const ri = STAGE_ORDER.indexOf(r._stage_sim);
      return ri > idx || r._stage_sim === 'Closed Won' || r._stage_sim === 'Closed Lost';
    }).length;
    return atOrPast > 0 ? past / atOrPast : null;
  }

  const avgDaysPerStage = { Prospecting:25, Qualification:20, Proposal:15, Negotiation:12 };
  function rateToDaily(stage, rate) {
    if (rate == null) return null;
    return +clamp(rate / avgDaysPerStage[stage], 0.001, 0.30).toFixed(4);
  }

  const prospAdv = advRate('Prospecting');
  const qualAdv  = advRate('Qualification');
  const propAdv  = advRate('Proposal');
  const negAdv   = advRate('Negotiation');

  // ---- Deal size → revenue lognormal params ----
  const amounts = oppsWithStage.map(r => r._amount).filter(a => a && a > 0);
  let rev_mu = null, rev_sigma = null, dealStats = null;
  if (amounts.length > 5) {
    const logAmts = amounts.map(a => Math.log(a));
    const mu = logAmts.reduce((s,v)=>s+v,0) / logAmts.length;
    const sigma = Math.sqrt(logAmts.map(v=>(v-mu)**2).reduce((s,v)=>s+v,0) / logAmts.length);
    rev_mu = +mu.toFixed(3);
    rev_sigma = +clamp(sigma, 0.2, 2.5).toFixed(3);
    amounts.sort((a,b)=>a-b);
    dealStats = {
      median: amounts[Math.floor(amounts.length/2)],
      p10: amounts[Math.floor(amounts.length*0.10)],
      p90: amounts[Math.floor(amounts.length*0.90)],
      mean: amounts.reduce((s,v)=>s+v,0)/amounts.length,
      n: amounts.length,
    };
  } else {
    warnings.push({ level:'warn', msg: 'No amount column or fewer than 5 valid values — revenue distribution not fitted.' });
  }

  // ---- Lead conversion ----
  let base_prob_lead_conversion = null;
  if (leads && leads.length > 0) {
    const convCol = Object.keys(leads[0]).find(k => ['isconverted','is_converted','converted'].includes(k.toLowerCase()));
    if (convCol) {
      const nConv = leads.filter(r => String(r[convCol]).toLowerCase() === 'true' || r[convCol] === '1').length;
      base_prob_lead_conversion = +clamp(nConv / leads.length, 0.01, 0.60).toFixed(4);
    }
  }

  if (!stageCol && !amounts.length) {
    warnings.push({ level:'error', msg: 'Could not fit any parameters. Make sure your CSV has at least a stage column and/or an amount column.' });
  }

  const fitted = {
    label: 'csv_fitted',
    n_reps: n_reps || 8,
    n_seed_accounts,
    days: 90,
    n_runs: 100,
    base_prob_close,
    base_prob_lost,
    base_prob_prospecting:    rateToDaily('Prospecting', prospAdv),
    base_prob_qualification:  rateToDaily('Qualification', qualAdv),
    base_prob_proposal:       rateToDaily('Proposal', propAdv),
    base_prob_negotiation:    rateToDaily('Negotiation', negAdv),
    base_prob_lead_conversion,
    inactivity_alpha,
    stagnation_alpha,
    apply_decay: true,
    notes: `Fitted from CSV export. ${oppsWithStage.length} opportunities, ${nClosed} closed (${nWon} won, ${nLost} lost).`,
  };

  const observed = {
    n_opps: oppsWithStage.length, n_closed: nClosed, n_won: nWon, n_lost: nLost,
    win_rate, lost_rate, avg_cycle,
    dealStats, stageCounts,
    adv_rates: { Prospecting: prospAdv, Qualification: qualAdv, Proposal: propAdv, Negotiation: negAdv },
    rev_mu, rev_sigma,
  };

  return { fitted, observed, warnings };
}

function drawCsvStageChart(stageCounts) {
  const canvas = document.getElementById('csv-stage-chart');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  const dpr = window.devicePixelRatio || 1;
  const W = canvas.offsetWidth || 400;
  const H = 160;
  canvas.width = W * dpr; canvas.height = H * dpr;
  ctx.scale(dpr, dpr);
  ctx.clearRect(0, 0, W, H);

  const stages = ['Prospecting','Qualification','Proposal','Negotiation','Closed Won','Closed Lost'];
  const colors = ['#378ADD','#BA7517','#1D9E75','#7F77DD','#0F6E56','#E24B4A'];
  const vals = stages.map(s => stageCounts[s] || 0);
  const maxVal = Math.max(...vals, 1);
  const barW = (W - 20) / stages.length - 6;
  const isDark = matchMedia('(prefers-color-scheme: dark)').matches;
  const textColor = isDark ? 'rgba(200,190,170,0.7)' : 'rgba(60,60,50,0.6)';

  vals.forEach((v, i) => {
    const x = 10 + i * ((W - 20) / stages.length);
    const barH = (v / maxVal) * (H - 50);
    ctx.fillStyle = colors[i];
    ctx.beginPath(); ctx.roundRect(x, H - 30 - barH, barW, barH, 3); ctx.fill();
    ctx.fillStyle = textColor;
    ctx.font = `10px -apple-system, sans-serif`; ctx.textAlign = 'center';
    ctx.fillText(v, x + barW/2, H - 32 - barH - 2);
    ctx.fillText(stages[i].replace(' ','\n').split(' ')[0], x + barW/2, H - 14);
  });
}
